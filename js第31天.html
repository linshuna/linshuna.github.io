鼠标滚轮放大缩小图片

    img.onmousewheel = function (e) {
    if (e.wheelDelta>0) {
        var change = 10;
    }else{
        var change = -10;
    }
    //计算鼠标在图片上的位置比例
    var scaleX = (e.clientX-img.offsetLeft)/img.offsetWidth;
    var scaleY = (e.clientY-img.offsetTop)/img.offsetHeight;    
    // 0/300 0
    // 150/300 0.5
    // 300/300 1
    //改变图片宽度
    img.style.width = img.offsetWidth + change + "px";
    //change*scaleX根据图片改变的大小值和鼠标在图片上的比例值计算该图片需要改变的位置
    img.style.left = img.offsetLeft - change*scaleX + "px";
    // 计算当图片宽度改变了高度改变了多少的值
    var changeH = change/(img.offsetWidth/img.offsetHeight);
    // 200/100 = 2  
    // 200+change/2 =100 +  changeH:5   
    img.style.top = img.offsetTop - changeH*scaleY+ "px";
    return false;
}

//拖拽滚轮
scrollbar.onmousedown = function (e) {
    var dy = e.clientY - scrollbar.offsetTop;
    document.onmousemove = function(e){
        var y = e.clientY - dy;
        move(y)
    }   
    return false;
}
scrollbar.onclick = function(e){
    //阻止冒泡
    e.cancelBubble = true;
}
document.onmouseup = function(){
    document.onmousemove = "";
}

//点击滚动条
scroll.onclick = function(e){
    var y = e.clientY-wrap.offsetTop-5-scrollbar.offsetHeight/2;    
    move(y)
}

function move(y){
    // 判断滚轮不能超出wrap框的顶部
    if (y<0) {y = 0}
    // 判断滚轮不能超出wrap框的底部
    if (y>scroll.offsetHeight-scrollbar.offsetHeight) {
        y = scroll.offsetHeight-scrollbar.offsetHeight;
    }
    scrollbar.style.top = y + "px";
    //获取内容超出部分的高度
    var conH = con.offsetHeight-wrap.clientHeight;
    //获取滚动条可以移动位置的高度
    var scrollH = scroll.offsetHeight-scrollbar.offsetHeight;
    // 计算内容移动位置的比例（根据内容和滚动条可移动的位置）
    var scale = conH/scrollH;
    con.style.top = -y*scale+"px";
}

// 添加滚轮事件
mousewheel(wrap,function(e,down){
    if (down) {
        var y = scrollbar.offsetTop+10;
    }else{
        var y = scrollbar.offsetTop-10;
    }
    move(y);
})

// 键盘上下控制滚轮移动
document.onkeydown = function(e){
    if (e.keyCode == 38) {
        var y = scrollbar.offsetTop-10;
    }else if(e.keyCode == 40){
        var y = scrollbar.offsetTop+10;
    }
    move(y)
}

//deviceorientation:设备方向
    window.addEventListener('deviceorientation',function (e) {      
        // alpha-对应CSS3的rotateZ
        // beta-对应CSS3的rotateX
        // gamma-对应CSS3的rotateY
        document.body.innerHTML = "alpha:"+parseInt(e.alpha) + "<br>beta:"+parseInt(e.beta) + "<br>gamma:"+parseInt(e.gamma)
    })

var bx = 0;
var by = 0;
var bz = 0; 
var btime = 0;
var num = 0;
//  devicemotion设备动作
if (window.DeviceMotionEvent){
    window.addEventListener("devicemotion",function (e) {
        // console.log(e);
        var motion = e.accelerationIncludingGravity;
        var x = parseInt(motion.x);
        var y = parseInt(motion.y);
        var z = parseInt(motion.z);
        document.body.innerHTML = "x:"+x+"<br>y:"+y+"<br>z:"+z;
        var now = new Date();
        if (now-btime>300) {
            num += Math.abs(x+y+z-bx-by-bz);
            if (num>1000) {
                alert("你成功摇到红包了");
            }
        }
        bx = x;
        by = y;
        bz = z;
        btime = now;
        
    })
}else{
    alert("设备不支持摇一摇功能！");
}
