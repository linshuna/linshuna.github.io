拖拽物体

注意点：该物体一定要定位，否则没有效果

方法1：用布尔值解决

var div = document.getElementsByTagName('div')[0];
var bol = false;
var dx = 0;
var dy = 0;
div.onmousedown = function (e) {
    bol = true;
    // 记录鼠标和DIV左上角位置的距离
    dx = e.clientX - div.offsetLeft;
    dy = e.clientY - div.offsetTop;
    return false;
}
document.onmousemove = function(e){
    if (!bol) {return}//判断没有按住鼠标就不能拖动
    var x = e.clientX - dx;
    var y = e.clientY - dy;
    div.style.left = x + "px";
    div.style.top = y + "px";
}
document.onmouseup = function(){
    bol = false;
}

方法2：不要布尔值

var div = document.getElementsByTagName('div')[0];
div.onmousedown = function (e) {
    // 记录鼠标和DIV左上角位置的距离
    var dx = e.clientX - div.offsetLeft;
    var dy = e.clientY - div.offsetTop;
    document.onmousemove = function(e){
        var x = e.clientX - dx;
        var y = e.clientY - dy;
        div.style.left = x + "px";
        div.style.top = y + "px";
    }
    return false;
}
document.onmouseup = function(){
    document.onmousemove = "";
}

表单事件

  • onfocus 获得焦点时触发的事件
  • onblur 失去焦点时触发的事件
  • onchange 当内容发生改变而且失去焦点的时候触发的事件
  • onselect 选中文本框里面的文字时触发的事件
  • onsubmit 到内容提交到form表单的时候触发的事件
  • onreset 内容重置的时候触发

注意点：onsubmit 和 onreset只能设置在form表单上，不能用在其他元素上

window事件

  • window.onload 页面加载完候触发的事件
  • window.onscroll 窗口的滚动条滚动的时候触发的事件
  • window.onresize 窗口大小改变时候触发的事件
window.onresize = function () {
    var winW = document.documentElement.clientWidth;
    if (winW<480) {
        document.body.style.background = "red";//document.body指向html的body上
    }else if(winW<640){
        document.body.style.background = "blue";
    }else if(winW<960){
        document.body.style.background = "yellow";
    }else{
        document.body.style.background = "transparent";
    }
}

键盘移动div

var div = document.getElementsByTagName('div')[0];
var x = 0;
var y = 0;
var speed = 100;//速度
var vx = 0;
var vy = 0;
setInterval(function(){
    x += vx;
    y += vy;
    div.style.left = x + 'px';
    div.style.top = y + 'px';
},200);
document.onkeydown = function (e) {
    if (e.keyCode==37) {//左
        vx = -speed;vy = 0;
    }else if(e.keyCode==38){//上
        vx = 0;vy = -speed;
    }else if(e.keyCode==39){//右
        vx = speed;vy = 0;
    }else if(e.keyCode==40){//下
        vx = 0;vy = speed;
    }
}

方块之间的碰撞(方块的拖拽和8个边之间的比较)

    var div = document.getElementsByTagName('div');

    div[0].onmousedown = function (e) {
    var dx = e.clientX - div[0].offsetLeft;
    var dy = e.clientY - div[0].offsetTop;

    document.onmousemove = function(e){
        var x = e.clientX - dx;
        var y = e.clientY - dy;
        div[0].style.left = x + "px";
        div[0].style.top = y + "px";

        var l1 = x;
        var r1 = x + div[0].offsetWidth;
        var t1 = y;
        var b1 = y + div[0].offsetHeight;

        var l2 = div[1].offsetLeft;
        var r2 = l2 + div[1].offsetWidth;
        var t2 = div[1].offsetTop;
        var b2 = t2 + div[1].offsetHeight;
