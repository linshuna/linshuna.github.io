canvas画饼图

  • 注意点:textBaseline的默认值是alphabetic
var canvas = document.querySelector('canvas');
var ctx = canvas.getContext('2d');
//数据
var arr =[
    {"title":"第一季度","num":"200","color":"red"},
    {"title":"第二季度","num":"100","color":"green"},
    {"title":"第三季度","num":"300","color":"blue"},
    {"title":"第四季度","num":"300","color":"yellow"},
    {"title":"第五季度","num":"300","color":"pink"}
];
//计算总数
var count = 0;
for (var i = 0;i < arr.length;i++) {
    count += Number(arr[i].num);//计算arr里面的num值
}
//圆的偏转的角度
var deg1 = 0;
var deg2 = 0;
for (var i = 0;i < arr.length;i++) {
    deg1 = deg2;//deg1总等于deg2的上一个值
    deg2 += arr[i].num/count*360;//计算所占比例的角度
    draw(deg1,deg2,arr[i].color);
    drawTitle(300,10+25*i,arr[i].color,arr[i].title);
}
//画圆
function draw(deg1,deg2,color){
    ctx.beginPath();
    ctx.fillStyle = color;
    ctx.moveTo(200,200);
    ctx.arc(200,200,100,Math.PI/180*deg1,Math.PI/180*deg2);
    ctx.fill();
    ctx.closePath();
}
//小方块加文字
function drawTitle(x,y,color,title){
    ctx.beginPath();
    ctx.fillStyle = color;
    ctx.fillRect(x,y,20,20);
    ctx.font="15px 黑体";
    ctx.textBaseline = "middle";//默认是alphabetic
    ctx.fillStyle = "black";
    ctx.fillText(title,x+30,y+10);//30是小矩形的宽高再加点间距  10是小矩形的宽度/2
    
    ctx.fill();
}

    canvas的事件isPointInPath

  • isPointInPath判断x,y的位置是否再当前绘制的路径上面

  • 如果出现多个矩形或是圆,只有最后的方块才能执行isPointInPath,解决方案是重新画,再用isPointInPath判断

  • 注意点:isPointInPath的返回值是false和true
