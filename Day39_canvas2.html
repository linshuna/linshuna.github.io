图形的组合方式

  • source-over：为属性globalCompositeOperation属性的默认值表示新图形覆盖在原图形之上
  • source-atop：只绘制新图形与原图形相重叠的部分，以及原图形其他部分
  • source-in：只绘制新图形与原图形重叠的部分，新图形与原图形其他部分变透明
  • source-out：只绘制新图形与原图形不重叠的部分；新图形与原图形其他部分变透明
  • destination-over：表示原图形覆盖在新图形之上destination-atop：只绘制原图形与新图形相重叠的部分，以及新图形其他部分
  • destination-in：只绘制原图形与新图形重叠的部分，原图形与新图形其他部分变透明
  • destination-out：只绘制原图形与新图形不重叠的部分；原图形与新图形其他部分变透明
  • lighter：原有图形与新图形都绘制，重叠部分加色处理
  • xor：只绘制原、新图形不重叠之处，其他做透明色
  • copy：只绘制新图形，覆盖掉原图

var wrap = document.querySelector(".wrap");
    var canvas = wrap.children[0];
    //设置canvas的实际宽度和高度
    canvas.width = wrap.clientWidth;
    canvas.height = wrap.clientHeight;
    var ctx = canvas.getContext("2d");
    //填充画布颜色
    ctx.fillStyle = "#ccc";
    ctx.fillRect(0,0,canvas.width,canvas.height);
    canvas.onmousedown = function(e){
        var disX = wrap.offsetLeft-1;//-1是减去边框
        var disY = wrap.offsetTop-1;
        ctx.globalCompositeOperation = "destination-out";
        ctx.beginPath();
        document.onmousemove = function(e){
            var x = e.clientX - disX;
            var y = e.clientY - disY;
            
            // ctx.arc(x,y,5,0,Math.PI*2);
            ctx.lineWidth = 10;
            ctx.lineCap = "round";
            ctx.lineTo(x,y);
            ctx.stroke();
        }
        return false;
    }
    document.onmouseup = function(){
        document.onmousemove = "";
    }

createPattern是图片的平铺

  • createPattern(img,type)
  • img：图片
  • type： 1) repeat整个画布平铺 2) repeat-x在x轴方向上平铺 3) repeat-y在y轴方向上平铺 4) no-repeat不平铺
  • 注意点:img是有onload事件 ##.clip()剪切
    var canvas = document.querySelector("canvas");
    var ctx = canvas.getContext("2d");
    var img = document.querySelector("img");
    ctx.rect(100,100,400,400);//画一个空心矩形
    ctx.clip();//剪切
    img.onload = function(){
        var c = ctx.createPattern(img,"repeat");
        ctx.fillStyle = c;
        ctx.fillRect(0,0,800,800);
    }

drawImage

  • 绘制图像：把图片绘制在画布上面drawImage(img,x,y)

    1)img：图片
    2)x：图片横坐标的起始值
    3)y：图片纵坐标的起始值

  • 绘制固定宽高的图像：drawImage(img,x,y,w,h)

    1)img：图片;
    2)x：图片横坐标的起始值;
    3)y：图片纵坐标的起始值; 4)w：在画布中图片显示的宽度;
    5)h：在画布中图片显示的高度;

  • 复制或放大图像局部：drawImage(img,x,y,w,h,rx,ry,rw,rh)

    1)img：图片 2)x、y：源图片的坐标
    3)w、h：在源图片上裁剪的宽、高 4)rx、ry：显示在画布中的坐标
    5)rw、rh：显示在画布中的宽、高

getImageData,putImageData和DataURL必须要再服务器上跑

  • getImageData 是对应rgba
  • getImageData的参数有4个,putImageData的参数有3(像drawImage)
  • .data是获取像素点数据
  • 注意点:颜色的取反 255-num 和 num 是相反颜色
  • 例如:rgb(0,0,0)--(黑色) 那么rgb(255,255,255)--(白色)
  • 把canvas里面的图片数据转成base64的地址数据格式 base64是可以加密也可以解密
    var btn = document.querySelector('button');

        //换为黑白色
        for (var i = 0;i<imageData.data.length; i+=4) {
            var c = imageData.data[i]*.3+imageData.data[i+1]*.59+imageData.data[i+2]*.11;//是黑白

            imageData.data[i+1] = c;
            imageData.data[i+2] = c;

        }-->
        
        ctx.putImageData(imageData,0,0);

        var url = canvas.toDataURL("image/png");
        console.log(url);
    }

function bg(){
    var deg = 0;
    var r = 150;
    ctx.beginPath();
    for (var i = 0;i < 13;i++) {
        // setInterval(function(){ //看动画效果
        deg += 30;
        var x = Math.sin(Math.PI/180*deg)*r;
        var y = -Math.cos(Math.PI/180*deg)*r;
        
        ctx.lineTo(x,y);

        // ctx.stroke();

    ctx.closePath();
    //填充颜色
    var c = ctx.createRadialGradient(0,0,0,0,0,150);
    c.addColorStop(0,"#22f");
    c.addColorStop(1,"#0ef");
    ctx.fillStyle = c;
    ctx.fill();
}

//画刻度
kedu();
function kedu(){
    var deg = 0;
    for (var i = 0;i < 60;i++) {
        ctx.save();
        ctx.rotate(Math.PI/30*i);
        ctx.beginPath();
        ctx.moveTo(0,-150);
        if (i%5==0) {
            ctx.lineWidth = 5;
            ctx.lineTo(0,-135);
        }else{
            ctx.lineWidth = 3;
            ctx.lineTo(0,-138);
        }   
        ctx.strokeStyle = "white";
        ctx.stroke();
        ctx.restore();
    }
}

//画数字
num();
function num(){
    for (var i = 1;i < 13;i++) {
        // ctx.save();
        // ctx.rotate(Math.PI/6*i);
        var x = Math.sin(Math.PI/6*i)*120;
        var y = -Math.cos(Math.PI/6*i)*120;
        ctx.font = "20px 黑体";
        ctx.textBaseline = "middle";
        ctx.textAlign = "center";
        ctx.fillStyle = "white";
        ctx.fillText(i,x,y+3);
        // ctx.restore();
    }
}

function sFn(deg){
    ctx.save();
    ctx.rotate(Math.PI/180*deg);
    ctx.beginPath();
    ctx.moveTo(0,5);
    ctx.lineTo(0,-120);
    ctx.strokeStyle = "white";
    ctx.stroke();
    ctx.closePath();
    ctx.restore();

function mFn(deg){
    ctx.save();
    ctx.rotate(Math.PI/180*deg);
    ctx.beginPath();
    ctx.moveTo(0,5);
    ctx.lineTo(0,-100);
    ctx.strokeStyle = "white";
    ctx.lineCap = "round";//让线的两边变为圆

    ctx.closePath();
    ctx.restore();
