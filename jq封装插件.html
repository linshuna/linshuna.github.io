留言板

注意点:如果是要创建多个标签,一定要加var,否则就变成全局变量(即只能创建一次)

需要引入jquery-3.0.0.min.js和动画效果的jquery.easing.min.js

var $name = $("#txt1");
    var $con = $("#txt2");
    var $ul = $("#message_text ul");
    var $btn = $("#btn");
    $btn.on("click",function(){
        var txt1 = $name.val();
        var txt2 = $con.val();
        if (txt1 == ""|| txt2 == "") {
            alert("请填写完整的信息");
            return;
        }
        var $li = $('<li><h3>'+txt1+'</h3><p>'+txt2+'<a href="###">删除</a></p></li>');//创建要用var,不然就是全局变量,只能创建一次
        $ul.prepend($li);

        $con.val("");
        $con.focus();
        $li.slideDown(800,"easeOutBounce");//检验是否是被隐藏

        var $a = $li.find("a");
        $a.on("click",function(){
            $li.slideUp(800,"easeOutBounce",function(){
                $li.remove();
            });
        })
    })

封装大图滚动

$.fn.extend({
    swiper:function(options){
        var $wrap = $(this);
        var $inner = $wrap.find(".inner");
        var $left = $wrap.find(".left");
        var $right = $wrap.find(".right");
        var $dd = $wrap.find(".dd");
        var timer = null;//自动播放的定时器

        var defaults = {//默认参数不开
            loop:false,
            btn:false,
            dd:false,
            autoPlay:0
        }
        //获取用户设置的参数 覆盖默认参数
        for (var key in options) {
            defaults[key] = options[key];
        }
        //根据设置的数据显示或隐藏左右按钮
        if (defaults.btn) {
            $left.show();
            $right.show();
        }else{
            $left.hide();
            $right.hide();
        }
        //根据设置的参数显示或隐藏切换点
        if (defaults.dd) {
            $dd.show();
        }else{
            $dd.hide();
        }


        var index = 0;
        var moveBol = false;//记录是否移动
        createD();
        //根据图片块的数量创建切换点
        function createD(){
            for (var i = 0;i < $inner.children().length;i++) {
                var $span = $("<span></span>");
                $dd.append($span);
                //tab切换
                $span.on("click",function(){
                    // //清空所有点的样式
                    // $dd.children().removeClass("select");
                    // //给当前点击的点加上选中样式
                    // $(this).addClass("select");
                    index = $(this).index();
                    //根据点击的第几个点移动inner到指定的位置
                    move(index);
                })
            }
            //默认给第一个点加上选中状态
            $dd.children().eq(0).addClass("select");
            if (defaults.loop) {
                //无缝滚动,复制第一张放到最后一张,用做无缝滚轮效果 ,加ture是深复制,不加的话只是复制标签
                $slide = $inner.children().eq(0).clone(true);
                $inner.append($slide);
            }
            
        }
        //下一张
        $right.on("click",function(){
            if (moveBol) {return;}//如果执行了return,那么下面就不能再执行了
            index++;
            
            // var n = defaults.loop?2:1;//判断是减1还是减2

            if (defaults.loop&&index>$inner.children().length-2) {//6-2最后还加了一张
                    // index = 0;
                $dd.children().removeClass("select");
                $dd.children().eq(0).addClass("select");

                moveBol = true;//记录动画开始

                //这时的index值是5
                $inner.animate({left:-index*$wrap.innerWidth()},function(){
                    $inner.css("left",0);
                    moveBol = false;//记录动画结束
                })
                index = 0;          
            }else{
                if (index>$inner.children().length-1) {
                    index = $inner.children().length-1;
                }
                move(index);
            }
            
        })
        //上一张
        $left.on("click",function(){
            if (moveBol) {return;}
            index--;
            // var n = defaults.loop?2:1;
            if (index<0) {
                if (defaults.loop) {
                    // index = $inner.children().length-1;
                    index = $inner.children().length-2;//4(倒数第二张) 小于0要显示倒数第二张
                    //直接跳到最后一张(最后一张是第一张的图)然后往上一张
                    $inner.css("left",-$wrap.innerWidth()*(index+1));
                }else{
                    index = 0;
                }
                
            }
            move(index);
        })

        function move(index){
            moveBol = true;//记录动画开始

            $dd.children().removeClass("select");
            $dd.children().eq(index).addClass("select");

            $inner.stop().animate({//stop()是清除定时器
                left:-$wrap.innerWidth()*index,
            },function(){
                moveBol = false;//记录动画结束
            })
        }
        //默认是0
        if (defaults.autoPlay>0) {
            autoPlay(defaults.autoPlay)
        }
        //主动播放函数
        function autoPlay(time){
            timer = setInterval(function(){
                $right.click();//模拟事件,执行下一张的功能
            },time)
        }

    }
})

jq动画的三种写法

  • 方法一:
    $("div").animate({
        width:200,
    },800,"easeInOutElastic")
  • 方法二:
$("div").animate({
        width:[200,"easeInOutBounce"],
        height:[200,"easeInOutElastic"]
    })
  • 方法三:
    $("div").animate({
        width:500,
    },{
        easing:"easeInOutBounce",
        duration:1000,      
        complete:function(){
            alert("完成")
        }
    })

jq动画的事件

  • show() hide()显示和隐藏,加了时间参数就会有动画效果 toggle()

  • 上滑(隐藏),下滑(显示)动画效果(改变高度) 会先判断是否隐藏 slideUp() slideDown()--必须是display:"none",才会有动画效果 slideToggle()

  • 渐隐渐显,改变透明度opacity 0-1 fadeIn() fadeOut() fadeToggle() fadeTo(0.5) //从当前的透明度渐变到0透明度为0.5的位置

  • 自定义动画 animate()

  • delay延时 stop停止动画 (clearInterval) finish()//直接完成动画

注意点:阻止冒泡 ev.stopPropagation()

delay:延时只能动画与动画之间 给其他效果是没用的

show和hide默认有动画效果"swing",也可以用linear

$(document).on("click",function(){
    $("div").animate({
        left:500
    },2000).delay().animate({top:100},1000)

})
