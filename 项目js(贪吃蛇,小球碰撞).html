小球的碰撞（中间是有障碍物）

如果小球是用p标签写的效果和div写小球的不一样（原因是p标签有默认的margin值，会影响效果）

解决方案是将p默认的margin值改为0 *{margin:0;padding:0;}

    //获取元素
    //var wrap = document.getElementById("wrap");//
    var div = wrap.children[0];
    var p = wrap.children[1];
    var vx = 5;//速度
    var vy = 5;
    var x = 0;//改变方向
    var y = 0;

    var timer = setInterval(function(){
        x+=vx;
        y+=vy;
        //获取8个值
        var l1 = x;
        var r1 = x+div.offsetWidth;
        var t1 = y;
        var b1 = y+div.offsetHeight;

        var l2 = p.offsetLeft;
        var r2 = l2+p.offsetWidth;
        var t2 = p.offsetTop;
        var b2 = t2+p.offsetHeight;
        //第三步,比较8个边的碰撞
        if (l1<=r2 && r1>=l2 && t1<=b2  && b1>=t2) {//判断已经撞到边
            //div右边碰撞p的左边
            if (r1-l2<Math.abs(vx) && vx>0) {
                vx*=-1;
                x = l2-div.offsetWidth;//强制转换(会碰到一点点)
            }
            //div左边碰撞p的右边
            if (r2-l1<Math.abs(vx) && vx < 0) {
                vx*=-1;
                x = r2;
            }
            //div的上边碰撞p的下边
            if (b2-t1<Math.abs(vy) && vy<0){
                vy*=-1;
                y = b2;
                
            }
            //div的下边碰撞p的上边
            if (b1-t2<Math.abs(vy) &&vy>0){
                vy*=-1;
                y = t2-div.offsetHeight;
            }
        } 
        //第一步
        if (x<0 || x>wrap.clientWidth-div.offsetWidth) {
            //会出现一点点的超出,下面是解决问题(强制转换) 碰到一点点边边
            if (vx>0) {//vx速度是正数
                x = wrap.clientWidth-div.offsetWidth;
            }else if (vx<0) {//vx速度是负数
                x = 0;
            }
            vx *= -1;
        }
        //第二步
        if (y<0 || y>wrap.clientHeight-div.offsetHeight) {
            if (vy>0) {
                y = wrap.clientHeight-div.offsetHeight;
            }else if (vy<0) {
                y = 0;
            }
            vy *= -1;
        }
        //判断4个边

        div.style.left = x+"px";
        div.style.top = y+"px";
    },30);

小方块的拖动事件

注意点：window的滚轮事件，有卷起的那部分，要加上来 div.style.left一定要加单位，div块也一定要是定位

var div = document.getElementsByTagName('div')[0];
var winW = window.innerWidth;//屏幕宽
var winH = window.innerHeight;//屏幕高
div.onmousedown = function (e) {
    var dx = e.clientX - div.offsetLeft;
    var dy = e.clientY - div.offsetTop;
    // 获取滚动条位置
    var sTop = document.documentElement.scrollTop || document.body.scrollTop;
    var sLeft= document.documentElement.scrollLeft || document.body.scrollLeft;
    document.onmousemove = function(e){
        var x = e.clientX - dx;
        var y = e.clientY - dy;
        if (x<sLeft) {
            x = sLeft;
        }else if(x>winW-div.offsetWidth+sLeft){
            x = winW-div.offsetWidth+sLeft;
        }

        if(y<sTop){
            y = sTop;
        }else if(y>winH-div.offsetHeight+sTop){
            y = winH-div.offsetHeight+sTop;
        }
        div.style.left = x + "px";
        div.style.top = y + "px";
    }

    return false;
}
document.onmouseup = function(){
    document.onmousemove = "";
}

window.onscroll = function(){
    // 获取滚动条位置
    var sTop = document.documentElement.scrollTop || document.body.scrollTop;
    var sLeft= document.documentElement.scrollLeft || document.body.scrollLeft;

    if (sTop>div.offsetTop) {
        div.style.top = sTop + "px";
    }else if(div.offsetTop>sTop+winH-div.offsetHeight){
        div.style.top = sTop+winH-div.offsetHeight + "px";
    }
    console.log(div.offsetTop,sTop+winH-div.offsetHeight)
    if (sLeft>div.offsetLeft) {
        div.style.left = sLeft + "px";
    }else if(div.offsetLeft>sLeft+winW-div.offsetWidth){
        div.style.left = sLeft+winW-div.offsetWidth + "px";
    }
}

changeFood()
var timer = setInterval(function () {

    y += vy;
    // 判断撞墙

    }else if(x>wrap.clientWidth-20){
        x = 0;

//创建蛇的身体
function createBody(){
    var div = document.createElement("div");
    wrap.appendChild(div);
}
//改变食物的位置
function changeFood(){
    // 计算一行的格子数量
    var numx = parseInt(wrap.clientWidth/20)
    var numy = parseInt(wrap.clientHeight/20)
    //随机食物位置
    var rndx = Math.floor(Math.random()*numx)*20;
    var rndy = Math.floor(Math.random()*numy)*20;

    for (var i = 0; i < div.length; i++) {
        if (rndx==div[i].offsetLeft&&rndy==div[i].offsetTop) {
            // 判断如果食物位置如果在蛇身上重新调用改变食物的位置函数
            return changeFood();
        }
    }
    p.style.left = rndx + "px";
    p.style.top = rndy + "px";
}
//吃到食物或者吃到自己
function check(){
    if (x==p.offsetLeft&&y==p.offsetTop) {
        changeFood()
        createBody()
    }
    for (var i = 4; i < div.length; i++) {
        if (x==div[i].offsetLeft&&y==div[i].offsetTop) {
            gameover();
        }
    }
}
function gameover(){
    clearInterval(timer);
    alert("游戏结束")

    switch(e.keyCode){
        case 37://左
            if(vx==0&&!bol){vx = -speed;vy = 0;}
        break;
        case 38://上
            if(vy==0&&!bol){vx = 0;vy = -speed;}

            if(vx==0&&!bol){vx = speed;vy = 0;}
        break;

            if(vy==0&&!bol){vx = 0;vy = speed;}
        break;
    }
